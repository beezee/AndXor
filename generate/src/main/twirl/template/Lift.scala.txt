  implicit def lifta@{f.index}(implicit M: Monoid[Prod]): Inj[Prod, F[@{f.focus}]] = {
    val (@{
  ((f.lefts.toList.paramList("a") :+ "_") ++
    f.rights.toList.paramList("a", f.lefts.length)).mkString(", ")}) =
        M.zero
    Inj.instance((@{
      ((f.lefts.toList.paramList("a") :+ "_") ++
        f.rights.toList.paramList("a", f.lefts.length)).mkString(", ")}))
  }
